Algoritmo E2_EJ9_Listas_bidimensionales_introduccion
	//Dada una tabla de 2 filas y 3 columnas con las ventas 
	//de dos vendedores en tres meses, calcula y muestra el 
	//total de ventas de cada vendedor (suma por filas).
	
	
	definir vendedores, sumav1, sumav2 como entero	 
	definir filasVEND, columnaXMES como entero
	
	dimension vendedores[2,3]
	
	sumav1=0 
	sumav2=0
	filasVEND = 0
	
	mientras filasVEND < 2 Hacer
		columnaXMES = 0
		
		mientras columnaXMES < 3 Hacer
			
			escribir 'NVenta:' filasVEND
			leer vendedores[filasVEND, columnaXMES]
			
		SI filasVEND = 0 Entonces
				sumav1= sumav1 + vendedores[filasVEND,columnaXMES]
			SINO 
			SI 	filasVEND = 1 Entonces
					sumav2= sumav2 + vendedores[filasVEND,columnaXMES]
			FinSi
		FinSi
		columnaXMES = columnaXMES + 1			
		FinMientras
		
		
		filasVEND = filasVEND + 1
	FinMientras
	
	Escribir 'sumav1 Y sumav2 RESPECTIVAMENTE son de: ' sumav1 ' y ' sumav2
	
FinAlgoritmo
